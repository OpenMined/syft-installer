üìö Usage Examples

Common patterns and use cases for different environments:

# Perfect for Jupyter notebooks and Google Colab
import syft_installer as si

# Simple installation with email prompt
installer = si.Installer()
installer.install()

# Pre-configure email to skip one prompt
installer = si.Installer(email="researcher@university.edu")
installer.install()  # Only prompts for OTP via widget
# For CI/CD and automated deployments
import syft_installer as si
import os

# Get OTP from external system (email API, etc.)
def get_otp_from_external():
    # Your OTP retrieval logic here
    return "ABCD1234"

# Fully automated installation
installer = si.Installer(
    email=os.getenv("SYFT_EMAIL"),
    headless=True
)
installer.request_otp()
otp = get_otp_from_external()
installer.verify_otp(otp)
installer.start_client(background=True)
# Advanced usage with custom configuration
import syft_installer as si

# Install with specific apps and debug mode
si.install(
    install_apps="dataroom,whisper,calculator",
    debug=True,
    install_mode="interactive"
)

# Download only (for custom setup)
si.install(install_mode="download-only")

# Setup only (binary already downloaded)
si.install(install_mode="setup-only")
